\creatureName{Drow}

\textit{Medium humanoid (elf), neutral evil}

\rule{3.5in}{1mm}

\attrib{Armor Class}{15 (chain shirt)}\\
% \hitpoints \damage and \hit macros allow for an accurate calculation (and
% display) of the dice to roll and the expected values therein. More detailed
% info in the macros.tex file. As an example, below:
% The below line will write '11 (2d8 + 2)'. The 2 (number of dice), 8 (number of
% sides of the dice), and 2 (the external modifier) are provided, and the
% expected value of 11 (what DnD5e uses for streamlining) is calculated)
\hitpoints{3}{8}{0}\\
\attrib{Speed}{30 ft.}

\rule{3.5in}{1mm}

% simply input the stat values (not modifiers) into the cooresponding slots and
% the macro will calculate and produce a stat-table with appropriate modifiers.
% They are to be entered in the following order: STR DEX CON INT WIS CHA.
\stats{10}{14}{10}{11}{11}{12}

\rule{3.5in}{1mm}

\attrib{Skills}{Perception +2, Stealth +4}\\
\attrib{Senses}{darkvision 120 ft., passive Perception 12}\\
\attrib{Languages}{Elvish, Undercommon}\\
\attrib{Challenge}{1/4 (50 XP)}

\rule{3.5in}{1mm}

\ability{Fey Ancestry}{The drow has advantage on saving throws against being
  charmed, and magic can't put the drow to sleep.}\\\bigskip
\ability{Innate Spellcasting}{The drow's spellcasting ability is Charisma (spell
  save DC 11). It can innately cast the following spells, requiring no material
  components:}\\\bigskip
At will: \textit{dancing lights}\\
1/day each: \textit{darkness, faerie fire}\\\bigskip
\ability{Sunlight Sensitivity}{While in sunlight, the drow has disadvantage on
  attack rolls, as well as on Wisdom (Perception) checks that rely on sight.}

\rule{3.5in}{1mm}

\vspace{5pt}

{\large\scshape{Actions}}

\vspace{-10pt}

\rule{3.5in}{0.5pt}

\action{Shortsword}{\mwa}{\toH{4}, \reach{5}, one target. 
  \damage{1}{6}{2}{Pie}}\\\bigskip
\action{Hand Crossbow}{\rwa}{\toH{4}, \rangeTwo{30}{120}, one target.
  \damage{1}{6}{2}{Pie} and the target must succeed on a DC 13 Constitution
  saving throw or be poisoned for 1 hour. If the saving throw fails by 5 or
  more, the target is also unconscious while poisoned in this way. The target
  wakes up if it takes damage or if another creature takes an action to wake it
  up.}
